# react+redux
---
# 项目介绍
  基于webpack+babel+react环境搭建
# 项目目录结构
  
  > dist //编译输出
  
  > node_modules //npm依赖包
  
  > --public        //公共静态资源
  >> --img //本地图片文件  
  
  > --scr 
    >> --index.html // 页面入口   
    >> --index.js // js入口  
  >> --page // 业务模块
  >>> --app.jsx  //页面子组件  
  >>> --app.scss //子组件样式 
  
  > .babelrc //babel配置文件  
  
  > package.json //项目配置信息 
  
  > webpack.config.js //webpack配置信息
  
# 依赖安装
 npm install
# 本地运行
npm run start
# 编译打包
npm run build
